Functor

class Functor f where
  fmap :: (a -> b) -> f a -> f b

Applicative

class Functor f => Applicative f where
  pure  :: a -> f a
  (<*>) :: f (a -> b) -> f a -> f b

> newtype Extents a = Extents {coordFromExtents :: Coord a} deriving (Eq, Ord)

> extentsFromCoord :: Num a => Coord a -> Extents a
